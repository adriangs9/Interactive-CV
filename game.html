let dragging = false; let activeKey = null; let offsetX = 0; let offsetY = 0; function svgPoint(evt) { const pt = svg.createSVGPoint(); pt.x = evt.clientX; pt.y = evt.clientY; return pt.matrixTransform(svg.getScreenCTM().inverse()); } function startDrag(e) { const key = e.currentTarget; const id = parseInt(key.dataset.key); if (solved || used.has(id)) return; dragging = true; activeKey = key; const pt = svgPoint(e); const t = key.transform.baseVal.consolidate(); const x = t ? t.matrix.e : 0; const y = t ? t.matrix.f : 0; offsetX = pt.x - x; offsetY = pt.y - y; key.classList.add("dragging"); } function drag(e) { if (!dragging || !activeKey) return; const pt = svgPoint(e); const x = pt.x - offsetX; const y = pt.y - offsetY; activeKey.setAttribute("transform", `translate(${x},${y})`); } function stopDrag() { if (!dragging || !activeKey) return; dragging = false; activeKey.classList.remove("dragging"); const id = parseInt(activeKey.dataset.key); checkCollision(activeKey, id); activeKey = null; } keys.forEach(k => k.addEventListener("mousedown", startDrag)); document.addEventListener("mousemove", drag); document.addEventListener("mouseup", stopDrag);
